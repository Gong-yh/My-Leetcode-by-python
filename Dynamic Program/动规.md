### E 509 斐波那契数列
#### 方法1：创建整个DP数组
```python
class Solution:
    def fib(self, n: int) -> int:
        '''
        动规解题五部曲：
            1. DP数
            2. 递推公式
            3. DP数组初始化
            4. 遍历顺序
            5. 出现问题，打印数组
        '''
        if n == 0:
            return 0
        Dp = [0 for _ in range(n+1)]
        Dp[1] = 1
        for i in range(2,n):
            Dp[i] = Dp[i-1] + Dp[i-2]
        return Dp[n-1] + Dp[n-2]
```

```python
class Solution:
    def minCostClimbingStairs(self, cost: List[int]) -> int:
        '''
        动规解题五部曲：
```

```python
class Solution:

        Dp = [0 for _ in range(len(cost))]

        Dp[0] = cost[0]
        Dp[1] = cost[1]
        for i in range(2,len(cost)):
            Dp[i] = min(Dp[i-1],Dp[i-2]) + cost[i]
            # print(Dp) 
        print(Dp)
        return min(Dp[-1],Dp[-2])
```

### M 62 找不同路径
```python
class Solution:
    def uniquePaths(self, m: int, n: int) -> int:
        # 前进方向右 || 下 （在前一步的基础上）
        # DP记录坐标（i,j） i or j +=1
        # 递归式：
            # f(n) = 右f(n-1)[0]+1,f(n-1) or 下f(n-1),f(n-1)+1
        
        # 怎么记录有多少条不同路径呢
        # 当前格子的路径是上面/左边格子+1 
        # F(m,n) = F(m-1,n) + F(m,n-1)
        Dp = [[1] for i in range(m)] #m*1
        Dp[0] = [1 for i in range(n)]
        # Dp[0].append(1)
        for i in range(1,m):
            # if i == 0:
                # Dp[i].append(Dp[i][j-1])
            for j in range(1,n):
                print(Dp)
                Dp[i].append(Dp[i][j-1]+Dp[i-1][j])
        print(Dp)
        return Dp[-1][-1]
```

```python
class Solution:
    def uniquePaths(self, m: int, n: int) -> int:
        ##优化
        # 用一维数组滚动解决：
        Dp = [1 for _ in range(n)]
```

### M 63 不同路径Ⅱ
```python 
class Solution(object):
    def uniquePathsWithObstacles(self, obstacleGrid):
        n = len(obstacleGrid)
        m = len(obstacleGrid[0])
        dp= [[0] * m for _ in range(n)]
        #判断终点是否有障碍物
        dp[-1][-1] = 1 if obstacleGrid[-1][-1] == 0 else 0
        #处理最后一列
        for i in range(n - 2, - 1, -1):
            if obstacleGrid[i][-1] == 1 or dp[i + 1][-1] == 0:
                dp[i][-1] = 0
            else:
                dp[i][-1] = 1
        #处理最后一行        
        for j in range(m - 2, -1, -1):
            if obstacleGrid[-1][j] == 1 or dp[-1][j + 1] == 0:
                dp[-1][j] = 0
            else:
                dp[-1][j] = 1
        for i in range(n - 2, -1, -1):
            for j in range(m - 2, -1, -1):
                #如果当前格子是障碍物
                if obstacleGrid[i][j] == 1:
                    dp[i][j] = 0
                #路径总数来自于下方(dp[i+1][j])和右方(dp[i][j+1])         
                else:
                    dp[i][j] = dp[i + 1][j] + dp[i][j + 1]
        return dp[0][0]
```
